# enhance the script by using DRY in pushall() and commit()
pushall() {
    git add .
    if [ "$1" != "" ]
    then
        git commit -m "$1"
    else
        git commit -m update
    fi
    git push
}

add() {
    if [ "$1" != "" ]
    then
        git add $1
    else
        git add .
    fi
}

commit() {
    if [ "$1" != "" ]
    then
        git commit -m "$1"
	printf $1
    else
        printf '%s ' 'please write a commit message'
        read reply
        if [ "$1" != "" ]
        then
            git commit -m "$reply"
        else
            git commit -m update
        fi
    fi
}
